1. сборка pytorch v2.8.0 для Pascal
git clone --branch v2.8.0 https://github.com/pytorch/pytorch.git

патч
pascal-support.patch

cd pytorch
git submodule update --init --recursive
export USE_CUDA=1
export CMAKE_CUDA_ARCHITECTURES="60;61"
export TORCH_CUDA_ARCH_LIST="6.0;6.1"
export CUDA_HOME=/usr/local/cuda-12.6
export PATH=$CUDA_HOME/bin:$PATH
export CUDACXX=$CUDA_HOME/bin/nvcc
export MAX_JOBS=3
export USE_FBGEMM=0
export USE_AVX512=0
export BUILD_TEST=0  
python setup.py install или python setup.py bdist_wheel --skip-build
